"""paved -- common paver tasks.
"""
from paver.easy import options, path, task, Bunch, sh, needs, environment
import json

__cwd__ = path('.').abspath()


options(
    paved = Bunch(
        cwd = __cwd__,
        
        clean = Bunch(
            patterns = ["*.pyc", "*~", "*.pyo", "*#", ".#*", "*.lock", "*.log*", "*.orig"],
            dirs = [__cwd__]
            ),
        ),
    )

__all__ = ['clean']

[docs]def clean(options, info):
    """Clean up extra files littering the source tree.

    options.paved.clean_dirs: directories to search recursively
    options.paved.clean_patterns: patterns to search for and remove
    """
    info("Cleaning patterns %s", options.paved.clean.patterns)
    for wd in options.paved.clean.dirs:
        info("Cleaning in %s", wd)
        for p in options.paved.clean.patterns:
            for f in wd.walkfiles(p):
                f.remove()



@task
def sloccount():
    '''Print "Source Lines of Code" using sloccount and writes into file.
    
    Export is hudson compatible: sloccount.sc
    
    requirements:
     - sloccount should be installed.
     - tee and pipes are used

    sloccount: http://www.dwheeler.com/sloccount/
    hudson: http://hudson-ci.org/
    sloccount plugin: http://wiki.hudson-ci.org/display/HUDSON/SLOCCount+Plugin
    '''
    sh('sloccount --wide --details {files} | tee sloccount.sc'.format(files=' '.join(options.sources)))

    '''print paver options.
    
    Prettified by json.
    `long_description` is removed
    '''
    def replace_set(obj):
        if isinstance(obj, dict):
            for k,v in obj.items():
                if isinstance(v, set):
                    obj[k]=list(v)
                replace_set(v)
                if k=='long_description':
                    obj[k]='[not displayed]'
                
        return obj
    x = replace_set(environment.options)

__all__ = ['sloccount', 'findimports','pyflakes', 'pychecker']


@task

[docs]def sloccount():
    '''Print "Source Lines of Code" using sloccount_ and writes into file.
    
    Export is hudson_ plugin_ compatible: sloccount.sc
    
    requirements:
     - sloccount_ should be installed.
     - tee and pipes are used

    .. _sloccount: http://www.dwheeler.com/sloccount/
    .. _hudson: http://hudson-ci.org/
    .. _plugin: http://wiki.hudson-ci.org/display/HUDSON/SLOCCount+Plugin
    '''
    packages = options.setup.packages
    sh('sloccount --wide --details {files} | tee sloccount.sc'.format(files=' '.join(packages)))
